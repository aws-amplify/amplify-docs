    func resendCode() {
        _ = Amplify.Auth.resendConfirmationCode(for: .email) {(result) in
            switch result {
            case .success(let deliveryDetails):
                print("Resend code send to - \(deliveryDetails)")
            case .failure(let error):
                print("Resend code failed with error \(error)")
            }
        }
    }
